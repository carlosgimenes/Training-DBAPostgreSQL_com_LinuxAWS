#A PG_SETTINGS TEM A VANTAGEM DE CAPTURAR INFORMAÇÕES
#DE TODOS OS ARQUIVOS DE CONFIGURAÇÃO
#EXEMPLO DE CONSULTA NA PG_SETTINGS
SELECT name, source, setting
FROM pg_settings
WHERE source != 'default'
AND source != 'override'
ORDER by 2, 1;
